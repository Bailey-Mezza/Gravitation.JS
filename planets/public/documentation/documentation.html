JS Physics Engine

Welcome! This guide helps you integrate the physics renderer and engine into your project.

Setup Usage Example API Reference 

// Example: Import & Initialize
import { PhysicsEngine } from './engine/physicsEngine.js';
import { Renderer } from './ui/renderer.js';

const canvas = document.getElementById('canvas');
const engine = new PhysicsEngine([], []);
const renderer = new Renderer(canvas, canvas.getContext('2d'), camera, { value: 1 });

function loop() {
  requestAnimationFrame(loop);
  engine.simulateStep();
  renderer.render([], [], null, false);
}
loop();
